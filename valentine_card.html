    <script>

        function moveNo() {

            let noButton = document.getElementById('no');

            let randomX = Math.floor(Math.random() * 200) - 100; // Move between -100 to 100 px

            let randomY = Math.floor(Math.random() * 100) - 50;

            noButton.style.transform = `translate(${randomX}px, ${randomY}px)`;

        }


        function growYes() {

            let yesButton = document.getElementById('yes');

            let currentSize = parseFloat(window.getComputedStyle(yesButton).fontSize);

            

            if (currentSize < 50) { // Prevents infinite growth

                yesButton.style.fontSize = (currentSize + 5) + 'px';

                yesButton.style.padding = (currentSize + 5) / 2 + 'px';

            }

        }


        function popConfetti() {

            confetti({

                particleCount: 200,

                spread: 100,

                origin: { y: 0.6 }

            });


            document.getElementById('question').innerText = "See you tomorrow babe ❤️";

        }


        // Balloon generator

        function createBalloon() {

            let balloon = document.createElement("div");

            balloon.className = "balloon";

            document.body.appendChild(balloon);


            let randomX = Math.random() * window.innerWidth;

            balloon.style.left = `${randomX}px`;

            balloon.style.animationDuration = `${Math.random() * 3 + 3}s`;


            setTimeout(() => {

                balloon.remove();

            }, 5000);

        }


        setInterval(createBalloon, 700);

    </script>
